          <div className="flex items-center gap-2">
            <Award className="text-blue-400 w-5 h-5 sm:w-6 sm:h-6" />
            <h1 className="text-lg sm:text-xl font-bold tracking-wider">QG DA PRF</h1>
          </div>
          <div className="flex items-center gap-2 bg-slate-800 px-3 sm:px-4 py-1 sm:py-1.5 rounded-full border border-slate-700">
            <Coins className="text-yellow-400 w-4 h-4 sm:w-5 sm:h-5" />
            <span className="font-bold text-yellow-400 text-sm sm:text-base">{coins}</span>
          </div>
        </div>

        <div className="max-w-6xl mx-auto px-4 flex gap-4 sm:gap-6 border-t border-slate-800 pt-2 overflow-x-auto no-scrollbar scroll-smooth">
          {[
            { id: 'revisoes', icon: Repeat, label: 'Revisões' },
            { id: 'ciclo', icon: BookOpen, label: 'Meu Ciclo' },
            { id: 'calendario', icon: CalendarIcon, label: 'Calendário' },
            { id: 'taf', icon: Dumbbell, label: 'Treino TAF' },
            { id: 'simulados', icon: Target, label: 'Simulados' },
            { id: 'estatisticas', icon: PieChart, label: 'Dashboard' },
            { id: 'loja', icon: Store, label: 'Loja' },
            { id: 'config', icon: Settings, label: 'Ajustes' }
          ].map(tab => (
            <button 
              key={tab.id} onClick={() => setActiveTab(tab.id)}
              className={`flex items-center gap-1.5 whitespace-nowrap pb-3 px-1 sm:px-2 font-semibold text-xs sm:text-sm transition-colors relative ${activeTab === tab.id ? 'text-blue-400' : 'text-slate-400 hover:text-slate-200'}`}
            >
              <tab.icon className="w-3.5 h-3.5 sm:w-4 sm:h-4" /> {tab.label}
              {activeTab === tab.id && <div className="absolute bottom-0 left-0 w-full h-1 bg-blue-500 rounded-t-md"></div>}
            </button>
          ))}
        </div>
      </header>

      <main className="max-w-6xl mx-auto mt-4 sm:mt-8 px-4">
        {activeTab === 'revisoes' && renderRevisoes()}
        {activeTab === 'ciclo' && renderCiclo()}
        {activeTab === 'calendario' && renderCalendar()}
        {activeTab === 'taf' && renderTAF()}
        {activeTab === 'simulados' && renderSimulados()}
        {activeTab === 'estatisticas' && renderEstatisticas()}
        {activeTab === 'loja' && renderLoja()}
        {activeTab === 'config' && renderConfig()}
      </main>

      {/* --- SISTEMA DE MODAIS TÁTICOS GLOBAIS --- */}

      {/* Modal de Alertas/Confirmações Seguras */}
      {globalModal && (
        <div className="fixed inset-0 bg-slate-900/80 flex items-center justify-center z-[80] p-4 fade-in backdrop-blur-sm">
          <div className="bg-white rounded-2xl w-full max-w-sm overflow-hidden shadow-2xl flex flex-col border-2 border-slate-200">
            <div className="p-6 text-center">
              <div className={`w-16 h-16 rounded-full flex items-center justify-center mx-auto mb-4 ${globalModal.isAlert ? 'bg-blue-100 text-blue-600' : 'bg-red-100 text-red-600'}`}>
                {globalModal.isAlert ? <Info className="w-8 h-8" /> : <AlertTriangle className="w-8 h-8" />}
              </div>
              <h3 className="text-xl font-black text-slate-800 mb-2">{globalModal.title}</h3>
              <p className="text-sm text-slate-600 leading-snug">
                {globalModal.message}
              </p>
            </div>
            <div className="bg-slate-50 p-4 flex gap-3 border-t border-slate-200">
              {!globalModal.isAlert && (
                <button onClick={() => setGlobalModal(null)} className="flex-1 py-3 font-bold text-slate-600 bg-white border border-slate-300 rounded-xl hover:bg-slate-100 transition-colors">Cancelar</button>
              )}
              <button onClick={() => { 
                if(globalModal.onConfirm) globalModal.onConfirm(); 
                setGlobalModal(null); 
              }} className={`flex-1 py-3 font-bold text-white rounded-xl shadow-md transition-colors ${globalModal.isAlert ? 'bg-blue-600 hover:bg-blue-700' : 'bg-red-600 hover:bg-red-700'}`}>
                {globalModal.isAlert ? "Ciente" : "Confirmar"}
              </button>
            </div>
          </div>
        </div>
      )}

      {classToConfirm && (
        <div className="fixed inset-0 bg-slate-900/80 flex items-center justify-center z-[60] p-4 fade-in backdrop-blur-sm">
          <form onSubmit={(e) => {
            e.preventDefault();
            let amount = parseInt(e.target.amount.value) || 1;
            amount = Math.min(amount, classToConfirm.maxClasses); // Blindagem matemática
            watchClass(classToConfirm.subjectId, classToConfirm.slotIndex, amount);
            setClassToConfirm(null);
          }} className="bg-white rounded-2xl w-full max-w-sm text-center overflow-hidden">
            <div className="p-6">
              <div className="w-12 h-12 bg-emerald-100 text-emerald-600 rounded-full flex items-center justify-center mx-auto mb-4"><PlayCircle/></div>
              <h3 className="text-lg font-black mb-1">Registro de Avanço</h3>
              <p className="text-slate-500 text-xs mb-4">Você está avançando no assunto <strong>{classToConfirm.topicName}</strong> da matéria <strong>{subjects[classToConfirm.subjectId]?.name}</strong>.</p>
              
              <div className="text-left bg-slate-50 p-4 rounded-lg border border-slate-200 mb-2">
                <label className="text-[10px] font-black text-slate-500 uppercase block mb-2">Aulas concluídas nesta sessão:</label>
                <div className="flex items-center gap-3">
                  <input name="amount" type="number" min="1" max={classToConfirm.maxClasses} defaultValue="1" className="w-full border-2 border-slate-300 p-3 rounded-lg font-black text-slate-800 outline-none focus:border-emerald-500 text-center text-xl transition-colors" />
                </div>
                <p className="text-[10px] text-slate-400 mt-2 font-bold text-center">Faltam {classToConfirm.maxClasses} aulas para finalizar a teoria e gerar o cartão FSRS.</p>
              </div>
            </div>
            <div className="bg-slate-50 p-4 flex gap-2 border-t border-slate-100">
              <button type="button" onClick={() => setClassToConfirm(null)} className="flex-1 py-3 font-bold text-slate-600 border bg-white rounded-xl hover:bg-slate-100 transition-colors">Ainda Não</button>
              <button type="submit" className="flex-1 py-3 font-bold text-white bg-emerald-500 rounded-xl hover:bg-emerald-600 shadow-md transition-colors">Confirmar Execução</button>
            </div>
          </form>
        </div>
      )}

      {replaceModal && (
        <div className="fixed inset-0 bg-slate-900/80 flex items-center justify-center z-[60] p-4 fade-in backdrop-blur-sm">
          <div className="bg-white rounded-2xl w-full max-w-sm overflow-hidden">
            <div className="p-6">
              <h3 className="text-lg font-black mb-2 flex items-center gap-2"><Shuffle className="w-5 h-5 text-blue-600"/> Substituir Vaga Tática</h3>
              <select 
                 value={selectedReplacement}
                 onChange={(e) => setSelectedReplacement(e.target.value)}
                 className="w-full border-2 p-3 rounded-lg font-bold text-sm outline-none focus:border-blue-500"
              >
                <option value="">- Escolha a Matéria -</option>
                {Object.values(subjects).filter(s => s.id !== replaceModal.oldSubjectId).map(s => (
                  <option key={s.id} value={s.id}>{s.name}</option>
                ))}
              </select>
            </div>
            <div className="bg-slate-50 p-4 flex gap-2">
              <button onClick={() => { setReplaceModal(null); setSelectedReplacement(''); }} className="flex-1 py-2.5 font-bold bg-white border rounded-xl text-sm hover:bg-slate-100 text-slate-600">Cancelar</button>
              <button onClick={() => { handleReplaceSubject(selectedReplacement); setSelectedReplacement(''); }} className="flex-1 py-2.5 font-bold text-white bg-blue-600 rounded-xl text-sm hover:bg-blue-700 shadow-md">Confirmar Troca</button>
            </div>
          </div>
        </div>
      )}

      {reviewModal && (
        <div className="fixed inset-0 bg-slate-900/80 flex items-center justify-center z-[60] p-4 fade-in backdrop-blur-sm">
          <form onSubmit={handleReviewSubmit} className="bg-white rounded-2xl w-full max-w-sm overflow-hidden shadow-2xl">
            <div className="p-6">
              <h3 className="text-lg font-black mb-1 flex items-center gap-2"><Activity className="w-5 h-5 text-red-500"/> Auditoria de Revisão FSRS</h3>
              <p className="text-slate-700 font-bold text-sm mb-3">{reviewModal.topicName}</p>
              
              {reviewModal.lastPerformance !== undefined && (
                <div className={`p-3 rounded-lg mb-4 text-xs font-bold border ${getThermometer(reviewModal.lastPerformance).color}`}>
                  Última revisão: Acertou {Math.round(reviewModal.lastPerformance * 100)}%. Vamos melhorar essa marca!
                </div>
              )}
              
              <div className="flex gap-4">
                <div className="flex-1">
                  <label className="text-xs font-bold text-slate-500">Qtd. Feitas</label>
                  <input name="total" type="number" min="1" required className="w-full mt-1 border-2 border-slate-200 p-3 rounded-lg text-center font-bold outline-none focus:border-red-400"/>
                </div>
                <div className="flex-1">
                  <label className="text-xs font-bold text-slate-500">Acertos</label>
                  <input name="correct" type="number" min="0" required className="w-full mt-1 border-2 border-slate-200 p-3 rounded-lg text-center font-bold text-emerald-600 outline-none focus:border-emerald-400"/>
                </div>
              </div>
            </div>
            <div className="bg-slate-50 p-4 flex gap-3 border-t border-slate-200">
              <button type="button" onClick={() => setReviewModal(null)} className="flex-1 py-2.5 font-bold text-slate-600 bg-white border border-slate-300 rounded-xl hover:bg-slate-100 transition-colors">Cancelar</button>
              <button type="submit" className="flex-1 py-2.5 font-bold text-white bg-red-500 rounded-xl hover:bg-red-600 shadow-md transition-colors">Salvar Desempenho</button>
            </div>
          </form>
        </div>
      )}

      {showSimuladoModal && (
        <div className="fixed inset-0 bg-slate-900/80 flex items-center justify-center z-[60] p-4 fade-in backdrop-blur-sm">
          <form onSubmit={(e) => {
            e.preventDefault();
            const dataVal = e.target.data.value;
            
            const b1C = parseInt(e.target.b1C.value) || 0;
            const b1E = parseInt(e.target.b1E.value) || 0;
            const b2C = parseInt(e.target.b2C.value) || 0;
            const b2E = parseInt(e.target.b2E.value) || 0;
            const b3C = parseInt(e.target.b3C.value) || 0;
            const b3E = parseInt(e.target.b3E.value) || 0;

            const b1Liquida = b1C - b1E;
            const b2Liquida = b2C - b2E;
            const b3Liquida = b3C - b3E;
            const liquida = b1Liquida + b2Liquida + b3Liquida;

            if (editingSimulado) {
              setSimulados(prev => prev.map(s => s.id === editingSimulado.id ? { ...s, data: dataVal, b1C, b1E, b2C, b2E, b3C, b3E, b1Liquida, b2Liquida, b3Liquida, liquida } : s));
            } else {
              setSimulados(prev => [...prev, { id: Date.now(), data: dataVal, b1C, b1E, b2C, b2E, b3C, b3E, b1Liquida, b2Liquida, b3Liquida, liquida }]);
            }
            setShowSimuladoModal(false);
            setEditingSimulado(null);
          }} className="bg-white rounded-2xl w-full max-w-md overflow-hidden shadow-2xl flex flex-col max-h-[90vh]">
            <div className="p-5 border-b border-slate-100">
              <h3 className="text-lg font-black flex items-center gap-2"><TargetIcon className="w-5 h-5 text-purple-600"/> {editingSimulado ? 'Corrigir Simulado' : 'Registro de Combate'}</h3>
              <p className="text-xs font-bold text-slate-500 mt-1">Insira seus erros e acertos segmentados. O sistema fará a auditoria de eliminação.</p>
            </div>
            
            <div className="p-5 overflow-y-auto custom-scrollbar flex flex-col gap-4">
              <div>
                <label className="text-xs font-bold text-slate-500 block mb-1">Data de Execução</label>
                <input name="data" type="text" required defaultValue={editingSimulado ? editingSimulado.data : new Date().toLocaleDateString('pt-BR')} className="w-full border-2 border-slate-200 p-2.5 rounded-lg font-bold text-slate-800 outline-none focus:border-purple-400"/>
              </div>

              <div className="bg-slate-50 p-4 rounded-xl border border-slate-200 shadow-sm">
                <div className="flex justify-between items-center mb-2">
                  <span className="text-xs font-black text-slate-700 uppercase">Bloco I - Básicas</span>
                  <span className="text-[10px] font-bold text-slate-400">Total: 55 | Min: 15</span>
                </div>
                <div className="flex gap-3">
                  <div className="flex-1"><label className="text-[10px] font-bold text-slate-400 uppercase">Certas</label><input name="b1C" type="number" min="0" max="55" required defaultValue={editingSimulado?.b1C ?? ''} className="w-full mt-1 border-2 border-slate-200 p-2 rounded-lg font-bold text-emerald-600 text-center outline-none focus:border-emerald-400"/></div>
                  <div className="flex-1"><label className="text-[10px] font-bold text-slate-400 uppercase">Erradas</label><input name="b1E" type="number" min="0" max="55" required defaultValue={editingSimulado?.b1E ?? ''} className="w-full mt-1 border-2 border-slate-200 p-2 rounded-lg font-bold text-red-500 text-center outline-none focus:border-red-400"/></div>
                </div>
              </div>

              <div className="bg-slate-50 p-4 rounded-xl border border-slate-200 shadow-sm">
                <div className="flex justify-between items-center mb-2">
                  <span className="text-xs font-black text-slate-700 uppercase">Bloco II - Trânsito</span>
                  <span className="text-[10px] font-bold text-slate-400">Total: 30 | Min: 10</span>
                </div>
                <div className="flex gap-3">
                  <div className="flex-1"><label className="text-[10px] font-bold text-slate-400 uppercase">Certas</label><input name="b2C" type="number" min="0" max="30" required defaultValue={editingSimulado?.b2C ?? ''} className="w-full mt-1 border-2 border-slate-200 p-2 rounded-lg font-bold text-emerald-600 text-center outline-none focus:border-emerald-400"/></div>
                  <div className="flex-1"><label className="text-[10px] font-bold text-slate-400 uppercase">Erradas</label><input name="b2E" type="number" min="0" max="30" required defaultValue={editingSimulado?.b2E ?? ''} className="w-full mt-1 border-2 border-slate-200 p-2 rounded-lg font-bold text-red-500 text-center outline-none focus:border-red-400"/></div>
                </div>
              </div>

              <div className="bg-slate-50 p-4 rounded-xl border border-slate-200 shadow-sm">
                <div className="flex justify-between items-center mb-2">
                  <span className="text-xs font-black text-slate-700 uppercase">Bloco III - Direito</span>
                  <span className="text-[10px] font-bold text-slate-400">Total: 35 | Min: 10</span>
                </div>
                <div className="flex gap-3">
                  <div className="flex-1"><label className="text-[10px] font-bold text-slate-400 uppercase">Certas</label><input name="b3C" type="number" min="0" max="35" required defaultValue={editingSimulado?.b3C ?? ''} className="w-full mt-1 border-2 border-slate-200 p-2 rounded-lg font-bold text-emerald-600 text-center outline-none focus:border-emerald-400"/></div>
                  <div className="flex-1"><label className="text-[10px] font-bold text-slate-400 uppercase">Erradas</label><input name="b3E" type="number" min="0" max="35" required defaultValue={editingSimulado?.b3E ?? ''} className="w-full mt-1 border-2 border-slate-200 p-2 rounded-lg font-bold text-red-500 text-center outline-none focus:border-red-400"/></div>
                </div>
              </div>
            </div>

            <div className="bg-slate-50 p-4 flex gap-3 border-t border-slate-200">
              <button type="button" onClick={() => { setShowSimuladoModal(false); setEditingSimulado(null); }} className="flex-1 py-3 font-bold text-slate-600 bg-white border border-slate-300 rounded-xl hover:bg-slate-100 transition-colors">Abortar</button>
              <button type="submit" className="flex-1 py-3 font-bold text-white bg-purple-600 rounded-xl hover:bg-purple-700 shadow-md transition-colors">Executar Auditoria</button>
            </div>
          </form>
        </div>
      )}

      {showTafModal && (
        <div className="fixed inset-0 bg-slate-900/80 flex items-center justify-center z-[60] p-4 fade-in backdrop-blur-sm">
          <form onSubmit={(e) => {
            e.preventDefault();
            const dataVal = e.target.data.value;
            const genero = e.target.genero.value;
            const isM = genero === 'masculino';
            
            const barra = parseFloat(e.target.barra.value) || 0;
            const abdominal = parseFloat(e.target.abdominal.value) || 0;
            const impulsao = parseFloat(e.target.impulsao.value) || 0;
            const corrida = parseFloat(e.target.corrida.value) || 0;
            const shuttle = parseFloat(e.target.shuttle.value) || 0;

            const ptsBarra = calcBarra(barra, isM);
            const ptsAbdominal = calcAbdominal(abdominal, isM);
            const ptsImpulsao = calcImpulsao(impulsao, isM);
            const ptsCorrida = calcCorrida(corrida, isM);
            const ptsShuttle = calcShuttle(shuttle, isM);
            
            const totalPts = ptsBarra + ptsAbdominal + ptsImpulsao + ptsCorrida + ptsShuttle;

            const payload = {
              data: dataVal, genero, barra, abdominal, impulsao, corrida, shuttle,
              ptsBarra, ptsAbdominal, ptsImpulsao, ptsCorrida, ptsShuttle, totalPts
            };

            if (editingTaf) {
              setTafHistory(prev => prev.map(s => s.id === editingTaf.id ? { ...s, ...payload } : s));
            } else {
              setTafHistory(prev => [...prev, { id: Date.now(), ...payload }]);
            }
            setShowTafModal(false);
            setEditingTaf(null);
          }} className="bg-white rounded-2xl w-full max-w-md overflow-hidden shadow-2xl flex flex-col max-h-[90vh]">
            <div className="p-5 border-b border-slate-100">
              <h3 className="text-lg font-black flex items-center gap-2"><Dumbbell className="w-5 h-5 text-orange-600"/> {editingTaf ? 'Editar TAF' : 'Registrar TAF'}</h3>
              <p className="text-xs font-bold text-slate-500 mt-1">Insira as marcas reais. O sistema fará a conversão oficial de pontos.</p>
            </div>
            
            <div className="p-5 overflow-y-auto custom-scrollbar flex flex-col gap-4">
              <div className="flex gap-3">
                <div className="flex-1">
                  <label className="text-xs font-bold text-slate-500 block mb-1">Data</label>
                  <input name="data" type="text" required defaultValue={editingTaf ? editingTaf.data : new Date().toLocaleDateString('pt-BR')} className="w-full border-2 border-slate-200 p-2.5 rounded-lg font-bold text-slate-800 outline-none focus:border-orange-400"/>
                </div>
                <div className="flex-1">
                  <label className="text-xs font-bold text-slate-500 block mb-1">Gênero</label>
                  <select name="genero" defaultValue={editingTaf ? editingTaf.genero : 'masculino'} className="w-full border-2 border-slate-200 p-2.5 rounded-lg font-bold text-slate-800 outline-none focus:border-orange-400">
                    <option value="masculino">Masculino</option>
                    <option value="feminino">Feminino</option>
                  </select>
                </div>
              </div>

              <div className="grid grid-cols-2 gap-3">
                <div>
                  <label className="text-[10px] font-bold text-slate-400 uppercase block mb-1" title="Masculino: Repetições | Feminino: Segundos de Isometria">Barra (Reps/Segs)</label>
                  <input name="barra" type="number" step="0.1" min="0" required defaultValue={editingTaf?.barra ?? ''} className="w-full border-2 border-slate-200 p-2 rounded-lg font-bold text-slate-700 outline-none focus:border-orange-400 text-center"/>
                </div>
                <div>
                  <label className="text-[10px] font-bold text-slate-400 uppercase block mb-1">Abdominal (Reps)</label>
                  <input name="abdominal" type="number" min="0" required defaultValue={editingTaf?.abdominal ?? ''} className="w-full border-2 border-slate-200 p-2 rounded-lg font-bold text-slate-700 outline-none focus:border-orange-400 text-center"/>
                </div>
                <div>
                  <label className="text-[10px] font-bold text-slate-400 uppercase block mb-1">Impulsão (Metros)</label>
                  <input name="impulsao" type="number" step="0.01" min="0" required defaultValue={editingTaf?.impulsao ?? ''} className="w-full border-2 border-slate-200 p-2 rounded-lg font-bold text-slate-700 outline-none focus:border-orange-400 text-center"/>
                </div>
                <div>
                  <label className="text-[10px] font-bold text-slate-400 uppercase block mb-1">Corrida (Metros)</label>
                  <input name="corrida" type="number" min="0" required defaultValue={editingTaf?.corrida ?? ''} className="w-full border-2 border-slate-200 p-2 rounded-lg font-bold text-slate-700 outline-none focus:border-orange-400 text-center"/>
                </div>
                <div className="col-span-2">
                  <label className="text-[10px] font-bold text-slate-400 uppercase block mb-1">Shuttle Run (Segundos)</label>
                  <input name="shuttle" type="number" step="0.01" min="0" required defaultValue={editingTaf?.shuttle ?? ''} className="w-full border-2 border-slate-200 p-2 rounded-lg font-bold text-slate-700 outline-none focus:border-orange-400 text-center"/>
                </div>
              </div>
            </div>

            <div className="bg-slate-50 p-4 flex gap-3 border-t border-slate-200">
              <button type="button" onClick={() => { setShowTafModal(false); setEditingTaf(null); }} className="flex-1 py-3 font-bold text-slate-600 bg-white border border-slate-300 rounded-xl hover:bg-slate-100 transition-colors">Abortar</button>
              <button type="submit" className="flex-1 py-3 font-bold text-white bg-orange-600 rounded-xl hover:bg-orange-700 shadow-md transition-colors">Avaliar Pontuação</button>
            </div>
          </form>
        </div>
      )}

      <style dangerouslySetInnerHTML={{__html: `
        .no-scrollbar::-webkit-scrollbar { display: none; }
        .no-scrollbar { -ms-overflow-style: none; scrollbar-width: none; }
        .custom-scrollbar::-webkit-scrollbar { height: 6px; width: 6px; }
        .custom-scrollbar::-webkit-scrollbar-track { background: transparent; }
        .custom-scrollbar::-webkit-scrollbar-thumb { background: #cbd5e1; border-radius: 4px; }
        .fade-in { animation: fadeIn 0.2s ease-in-out; }
        @keyframes fadeIn { from { opacity: 0; transform: translateY(10px); } to { opacity: 1; transform: translateY(0); } }
      `}} />
    </div>
  );
}
